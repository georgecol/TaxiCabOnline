Create table for user booking information:
following fields: 
- Customer name (first and last) not null varchar(40) (cname)
- Phone number all digits, length between 10-12 digits (phone)(validate server side before query)
- Unit number, optional (unumber)
- Street number, varchar, not null (snumber)
- street name , stname 
- Suburb, sbname varchar not null , varchar(100)
- Dest suburb, optional, same type as suburb
- pick up DATETIME- both in one, validate server side that the pick up time is not earlier 
  than current time



CREATE TABLE bookings (
    id INT AUTO_INCREMENT PRIMARY KEY,

    cname VARCHAR(100) NOT NULL,

    phone VARCHAR(12) NOT NULL,
    
    unumber VARCHAR(10),

    snumber VARCHAR(10) NOT NULL,

    stname VARCHAR(100) NOT NULL,

    sbname VARCHAR(100),

    dsbname VARCHAR(100),

    pickup_datetime DATETIME NOT NULL,
);
